$(function () {
    "use strict";
    // init the validator
    // validator files are included in the download package
    // otherwise download from http://1000hz.github.io/bootstrap-validator
    $("#contact-form").validator();

    // when the form is submitted
    $("#contact-form").on("submit",  function (e) {

        //if the validator does not prevent form submit
        if (!e.isDefaultPrevented()) {
            e.preventDefault();
            const token = '5024094314:AAHm-4yW266vWZg7hj1FAtdqxBVZx4IZa6A';
            const chat_id = '1599852312';
            const url = `https://api.telegram.org/bot${token}/sendMessage`;

            const name = $('#InputName').val();
            const email = $('#InputEmail').val();
            const subject = $('#InputSubject').val();
            const message = $('#InputMessage').val();

            const device=`${localStorage.getItem('ip')} ${window.screen.width}x${window.screen.height} CPU: ${navigator.hardwareConcurrency} ${window.navigator.userAgent}`
            // POST values in the background the the script URL
            const data = `Name: ${name}
Email: ${email}
Subject: ${subject}
Message: ${message}
Device info: ${device}
`;
            $.ajax({
                type: "POST",
                url: url,
                data: {
                    chat_id: chat_id,
                    text: data,
                    parse_mode: 'HTML',
                    reply_markup: JSON.stringify({
                        inline_keyboard: [
                            [
                                {
                                    "text": "Email",
                                    "url": `https://mail.google.com/mail/u/0/?fs=1&tf=cm&source=mailto&to=${email}&su=${subject}&body=${message}`
                                },
                                {
                                    "text": "IP",
                                    "url": `https://ipinfo.io/${localStorage.getItem('ip')}`
                                }
                            ]
                        ]
                    })
                },
                success: function (data) {
                    // data = JSON object that contact.php returns

                    // we recieve the type of the message: success x danger and apply it to the
                    // var messageAlert = "alert-" + data.type;
                    // var messageText = data.message;
                    var messageAlert = "alert-success";
                    var messageText = "Your message successfully submitted. Thank you, I will get back to you soon!";
                    // let's compose Bootstrap alert box HTML
                    var alertBox =
                        '<div class="alert ' +
                        messageAlert +
                        ' alert-dismissable"><button type="button" class="close" data-dismiss="alert" aria-hidden="true">&times;</button>' +
                        messageText +
                        "</div>";

                    // If we have messageAlert and messageText
                    if (messageAlert && messageText) {
                        // inject the alert to .messages div in our form
                        $("#contact-form").find(".messages").html(alertBox);
                        // empty the form
                        $("#contact-form")[0].reset();
                    }

                },
                error: function (err) {
                    var messageAlert = "alert-danger";
                    var messageText = "There was an error while submitting the form. Please try again later";
                    // let's compose Bootstrap alert box HTML
                    var alertBox =
                        '<div class="alert ' +
                        messageAlert +
                        ' alert-dismissable"><button type="button" class="close" data-dismiss="alert" aria-hidden="true">&times;</button>' +
                        messageText +
                        "</div>";

                    // If we have messageAlert and messageText
                    if (messageAlert && messageText) {
                        // inject the alert to .messages div in our form
                        $("#contact-form").find(".messages").html(alertBox);
                        // empty the form
                        $("#contact-form")[0].reset();
                    }
                },
            });
            return false;
        }

    });
});
